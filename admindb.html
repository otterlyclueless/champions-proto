First-Time Setup Required

Run this SQL in your Supabase SQL Editor to enable write access from this dashboard:

-- 014_admin_write_access.sql
-- Allows authenticated users to update reference tables

CREATE POLICY "admin_update_pokemon" ON pokemon
  FOR UPDATE TO authenticated USING (true) WITH CHECK (true);

CREATE POLICY "admin_update_moves" ON moves
  FOR UPDATE TO authenticated USING (true) WITH CHECK (true);

CREATE POLICY "admin_update_abilities" ON abilities
  FOR UPDATE TO authenticated USING (true) WITH CHECK (true);

CREATE POLICY "admin_update_items" ON items
  FOR UPDATE TO authenticated USING (true) WITH CHECK (true);

CREATE POLICY "admin_update_pokemon_moves" ON pokemon_moves
  FOR UPDATE TO authenticated USING (true) WITH CHECK (true);
